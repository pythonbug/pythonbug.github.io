```
;编程，用加法和移位指令计算(ax)=(ax)*10
;提示，(ax)*10=(ax)*2+(ax)*8

;分析，很显然这里的乘法用位移
assume cs:code

code segment

    start:  mov ax,4        ;方便验证结果
            mov bx,ax       ;分两次位移
            shl ax,1        ;(ax)*2

            mov cl,3
            shl bx,cl       ;(ax)*8

            add ax,bx       ;此处debug查看一下ax的值，验证移位的正确性

            mov ax,4c00h
            int 21h

code ends

end start
```
