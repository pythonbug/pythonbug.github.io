# [](#header-1)要求
编程，将data段中的数据按如下格式写入到table段中，并计算21年中的人均收入（取整），结果也按照下面的格式保存在table中

## 完整代码：
```assembly
;实验7 寻址方式在结构化数据访问中的应用
;编程，将data段中的数据按如下格式写入到table段中，并计算21年中的人均收入（取整），
;结果也按照下萌的格式保存在table段中
assume cs:codesg
	data segment
		db '1975','1976','1977','1978','1979','1980','1981','1982','1983'
		db '1984','1985','1986','1987','1988','1989','1990','1991','1992'
		db '1993','1994','1995'
		;以上表示21年的21个字符串
		
		dd 16,22,382,1356,2390,8000,16000,24486,50065,97479,140417,197514
		dd 345980,590827,803530,1183000,1843000,2759000,3753000,4649000,5937000
		;以上表示21年公司总收入的21个dword型数据
		
		dw 3,7,9,13,28,38,130,220,476,778,1001,1442,2258,2793,4037,5635,8228
		dw 11542,14430,15257,17800
		;以上是表示21年公司雇员人数的21个word型数据
	data ends
	
	table segment
		db 21 dup('year summ ne ?? ')
	table ends
	
	codesg segment
	
		;初始化数据栈，要录入的内容放在ds中
		start:
		mov ax,data
		mov ds,ax
		
		;初始化table栈
		mov ax,table
		mov es,ax
		
		;初始化用到的寄存器
		mov bx,0
		mov si,0
		mov bp,0
	
		;初始化循环次数，一共是21年
		mov cx,21
		
		;年
		s:
		mov ax,[bx]
		mov dx,[bx+2]
		mov es:[si],ax
		mov es:[si+2],dx
		
		;收入
		mov ax,[bx+84]
		mov dx,[bx+86]
		mov es:[si+5],ax
		mov es:[si+7],dx
		
		;人均收入
		div word ptr ds:[bp+168]
		mov es:[si+0dh],ax
		
		;雇员
		mov ax,ds:[bp+168]	;bp如果不写清楚，默认是ss的段
		mov es:[0ah+si],ax
		
		;si表示table的行，bx表示数据项，bp是雇员和收入的数据项
		add si,10h
		add bx,4
		add bp,2
		loop s
	
		mov ax,4c00h
		int 21h
	codesg ends
end start
```
